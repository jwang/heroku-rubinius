EventMachine::Protocols::ObjectProtocol

ObjectProtocol allows for easy communication using marshaled ruby objects 

 module RubyServer
   include EM::P::ObjectProtocol

   def receive_object obj
     send_object({'you said' => obj})
   end
 end
